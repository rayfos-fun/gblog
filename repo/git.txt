.gitignore
  venv/

git add .  # to add all files into stage 
git add $FILE  # to add the specific file into stage
git branch  # show all branches
git branch $BRANCH  # create branch and migrate all existing commits into it (but stay in current branch)
git checkout $BRANCH  # switch to branch
git checkout -b $BRANCH   # create and switch to branch
git checkout --ours xxx  # override using ours (current branch) to resolve conflict
git clone https://github.com/rayfos-fun/gblog.git   # -b branch_name
git commit --amend  # to add touched files into current commit
git config --global push.autoSetupRemote true  # "git push" in branch could work without "-u origin ..."
git diff --staged  # to show diff of the commit before and after
git fetch  # sync origin status to local
git log  # show logs
  --graph --oneline --all --decorate
git pull   # git fetch + git merge
git push -u origin feature/comments  # origin: cloud repo e.g. github; -u=--set-upstream: track local feature/comments with origin (same name feature/comments)
git rebase main  # cut this branch, rebase it to main
  -i HEAD~2  # -i: interactive; pick -> reword => edit commit message
git reset  # to undo git add .
git reset --hard HEAD~1  # --hard: !!discard!! all changes
git reset --soft HEAD~2  # to undo 2 unpushed commits
git restore .  # to restore unstaged files to unmodified
  # --staged  to restore staged files to modified (staging area => working directory)
  # --source=$COMMIT  to restore commited files to staged (local repository => staging area)
git stash  # save current changes temporarily
git stach pop  # load previous stashed changes
git switch -c $BRANCH  # witch to branch; -c: create


# terminology

init => edit => modified (existing file) or untracked (new file) @[working directory]
=> git add ... => staged @[staging area]
=> git commit ... => commited @[local repository]
=> git push ... => pushed / synced @[remote repository] 

# scenario

## init
git clone https://github.com/rayfos-fun/gblog.git
cd gblog

## edit on main, no other branches
git add .
git commit -m 'commit meesage'
git push

## squash (all commits in branch) and merge (into single pushed commit) via Github
git switch -c feature/xyz
repeat git add .; git commit -m 'commit meesage'; 
git rebase main  # optional; to keep main in sync
Github => pull request => squash and merge => delete branch
git switch main
git pull
git branch -D feature/xyz

## work on local branch (branch not push)
git switch -c feature/xyz
git add .; git commit -m 'commit meesage'; ...
git switch main
git pull origin main  # or git pull; to make sure main is up-to-date
git merge feature/xyz
git push origin main  # or git push
git branch -d feature/xyz

## cowork on branch
git checkout -b feature/xyz
git add .; git commit -m 'commit meesage'; git push -u origin feature/xyz
goto github => pull request => merge
